// it must support rank, progress and the incProgress(rank) method

class User {
  constructor() {
    this._rank = -8;
    this._progress = 0;
  }
  
  get rank() {
    return this._rank;
  }
  
  get progress() {
    return this._progress;
  }
  
  _addProgress(points) {
    const allProgress = this._progress + points;
    const ranksToAdd = Math.floor(allProgress / 100);
    console.log(ranksToAdd);
    const restProgress = allProgress - ranksToAdd * 100;
    this._progress = restProgress;
    this._rank += ranksToAdd;
    if (this._rank === 0) this._rank++;
  }
  
  incProgress(rank) {
    if (rank > 8 || rank < -8 || rank === 0) throw new Error("Invalid rank");
    let diff = rank - this._rank;
    if (diff === 0) this._addProgress(3);
    else if (diff === -1) this._addProgress(1);
    else if (diff > 0) {
      this._addProgress(10*diff*diff);
    }
  }
}
